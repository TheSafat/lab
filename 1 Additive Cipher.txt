#include<bits/stdc++.h>
using namespace std;
string cipher(string s,int k)
{
    for(int i=0 ; i < s.length() ; i++){
        s[i] = ((s[i]-'a') + k) % 26 + 'A';
    }
    return s;
}
string decipher(string s,int k)
{
    for(int i=0 ; i < s.length() ; i++){
        s[i] = ((s[i]-'A') + (26-k)) % 26 + 'a';
    }
    return s;
}
int main()
{
    string p;
    int k;
    cin>>p>>k;
    string c = cipher(p,k);
    cout<< decipher(c,k) <<endl;
}
