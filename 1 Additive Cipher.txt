#include<bits/stdc++.h>
using namespace std;

string cipher(string p, int k)
{
    string c="";
    for(int i=0;i<p.size();i++) 
        c += (p[i]-'a'+k)%26+'A';
    return c;
}
string decipher(string c, int k)
{
    string p="";
    for(int i=0;i<c.size();i++)
        p += (c[i]-'A'-k+26)%26+'a';
    return p;
}

int main()
{
    cout<<decipher("HELLO",0);
    //cout<<encryption_additive("hello",0)<<endl;
}
